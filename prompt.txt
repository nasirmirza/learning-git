Option Explicit

' ===== CONFIGURATION MODULE =====
' Define all column mappings in one place for easy maintenance

Private Type ColumnMapping
    destCol As Integer          ' Destination column number
    destHeader As String        ' Destination column header
    mappingType As String       ' "DIRECT", "STATIC", "CONCAT", "FORMULA", "CUSTOM"
    sourceCol As Variant        ' Source column(s) - can be integer or array
    staticValue As String       ' For static mappings
    CustomFunction As String    ' Name of custom function to call
End Type

Private mappings() As ColumnMapping
Private mappingCount As Integer

' ===== MAIN TRANSFORMATION SUBROUTINE =====
Sub CreateTransformedSheet()
    Dim wsSource As Worksheet
    Dim wbNew As Workbook
    Dim wsDestination As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim j As Integer
    Dim savePath As String
    Dim fileName As String
    Dim startTime As Double
    
    startTime = Timer
    
    ' Set source worksheet
    Set wsSource = ActiveSheet
    
    ' Validate source sheet has headers
    If WorksheetFunction.CountA(wsSource.Rows(1)) = 0 Then
        MsgBox "Source sheet appears to have no headers!", vbExclamation
        Exit Sub
    End If
    
    ' Initialize column mappings
    Call InitializeColumnMappings
    
    ' Find last row with data in source sheet
    lastRow = wsSource.Cells(wsSource.Rows.Count, "A").End(xlUp).Row
    
    If lastRow < 2 Then
        MsgBox "No data found in source sheet!", vbExclamation
        Exit Sub
    End If
    
    ' Create new workbook
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    
    Set wbNew = Workbooks.Add
    Set wsDestination = wbNew.Worksheets(1)
    wsDestination.Name = "Transformed Data"
    
    ' Create headers in destination sheet
    For j = 1 To mappingCount
        wsDestination.Cells(1, mappings(j).destCol).Value = mappings(j).destHeader
    Next j
    
    ' Transform data row by row
    For i = 2 To lastRow
        Call TransformRow(wsSource, wsDestination, i)
        
        ' Progress indicator for large datasets
        If i Mod 1000 = 0 Then
            Application.StatusBar = "Processing row " & i & " of " & lastRow
        End If
    Next i
    
    ' Format headers
    With wsDestination.Range(wsDestination.Cells(1, 1), wsDestination.Cells(1, mappingCount))
        .Font.Bold = True
        .Interior.Color = RGB(68, 114, 196)
        .Font.Color = RGB(255, 255, 255)
        .HorizontalAlignment = xlCenter
    End With
    
    ' Auto-fit columns
    wsDestination.Cells.Columns.AutoFit
    
    ' Add filters
    wsDestination.Range(wsDestination.Cells(1, 1), wsDestination.Cells(1, mappingCount)).AutoFilter
    
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
    Application.StatusBar = False
    
    ' Prompt user to save the new file
    fileName = "Transformed_" & Format(Now, "yyyymmdd_hhmmss") & ".xlsx"
    savePath = Application.GetSaveAsFilename( _
        InitialFileName:=fileName, _
        FileFilter:="Excel Files (*.xlsx), *.xlsx", _
        Title:="Save Transformed Data")
    
    If savePath <> "False" Then
        wbNew.SaveAs fileName:=savePath, FileFormat:=xlOpenXMLWorkbook
        MsgBox "Transformation complete!" & vbCrLf & _
               "Rows processed: " & (lastRow - 1) & vbCrLf & _
               "Time taken: " & Format(Timer - startTime, "0.00") & " seconds" & vbCrLf & _
               "File saved as: " & savePath, vbInformation
    Else
        MsgBox "File not saved. The new workbook is still open but unsaved.", vbExclamation
    End If
    
End Sub

' ===== COLUMN MAPPING CONFIGURATION =====
' This is where you define all your mappings
Private Sub InitializeColumnMappings()
    mappingCount = 0
    ReDim mappings(1 To 100) ' Adjust size based on destination columns
    
    ' ===== DIRECT MAPPINGS (One-to-One) =====
    ' Syntax: AddMapping DestCol, "Header", "DIRECT", SourceCol
    
    Call AddMapping(5, "D5", "DIRECT", 1)      ' S1 ? D5
    Call AddMapping(4, "D4", "DIRECT", 2)      ' S2 ? D4
    Call AddMapping(1, "D1", "DIRECT", 5)      ' S5 ? D1
    
    ' Add more direct mappings:
    ' Call AddMapping(10, "Customer_Name", "DIRECT", 8)
    ' Call AddMapping(15, "Order_Date", "DIRECT", 12)
    
    
    ' ===== STATIC VALUE MAPPINGS =====
    ' Syntax: AddMapping DestCol, "Header", "STATIC", , "StaticValue"
    
    Call AddMapping(6, "D6", "STATIC", , "NEW")
    Call AddMapping(7, "D7", "STATIC", , "CRED")
    
    ' Add more static mappings:
    ' Call AddMapping(20, "Status", "STATIC", , "ACTIVE")
    ' Call AddMapping(25, "Source_System", "STATIC", , "SAP")
    
    
    ' ===== CONCATENATION MAPPINGS =====
    ' Syntax: AddMapping DestCol, "Header", "CONCAT", Array(Col1, Col2, Col3)
    
    Call AddMapping(2, "D2", "CONCAT", Array(3, 4))  ' S3 & S4 ? D2
    
    ' Add more concatenation mappings:
    ' Call AddMapping(30, "Full_Address", "CONCAT", Array(10, 11, 12, 13))
    ' Call AddMapping(35, "Full_Name", "CONCAT", Array(5, 6, 7))
    
    
    ' ===== FORMULA MAPPINGS =====
    ' Syntax: AddMapping DestCol, "Header", "FORMULA", SourceCol, FormulaPattern
    ' Use {SRC} as placeholder for source value
    
    Call AddMapping(8, "D8_Calculated", "FORMULA", 1, "={SRC}*1.1")
    
    ' Add more formula mappings:
    ' Call AddMapping(40, "Tax_Amount", "FORMULA", 15, "={SRC}*0.18")
    ' Call AddMapping(45, "Discounted_Price", "FORMULA", 20, "={SRC}*0.9")
    
    
    ' ===== CUSTOM FUNCTION MAPPINGS =====
    ' Syntax: AddMapping DestCol, "Header", "CUSTOM", , , "FunctionName"
    ' You'll define the custom function separately
    
    Call AddMapping(3, "D3", "CUSTOM", , , "CustomTransform_D3")
    
    ' Add more custom mappings:
    ' Call AddMapping(50, "Risk_Category", "CUSTOM", , , "CalculateRiskCategory")
    ' Call AddMapping(55, "Priority_Score", "CUSTOM", , , "CalculatePriority")
    
    
    ' ===== CONDITIONAL MAPPINGS =====
    ' For complex conditional logic, use CUSTOM type
    
    Call AddMapping(9, "D9_Conditional", "CUSTOM", , , "ConditionalMapping_D9")
    
End Sub

' Helper function to add mappings
Private Sub AddMapping(destCol As Integer, destHeader As String, _
                       mappingType As String, _
                       Optional sourceCol As Variant, _
                       Optional staticValue As String = "", _
                       Optional customFunc As String = "")
    mappingCount = mappingCount + 1
    
    With mappings(mappingCount)
        .destCol = destCol
        .destHeader = destHeader
        .mappingType = mappingType
        .sourceCol = sourceCol
        .staticValue = staticValue
        .CustomFunction = customFunc
    End With
End Sub

' ===== ROW TRANSFORMATION LOGIC =====
Private Sub TransformRow(wsSource As Worksheet, wsDestination As Worksheet, rowNum As Long)
    Dim j As Integer
    Dim sourceValue As Variant
    Dim resultValue As Variant
    Dim cols As Variant
    Dim k As Integer
    
    For j = 1 To mappingCount
        resultValue = ""
        
        Select Case mappings(j).mappingType
            
            Case "DIRECT"
                ' Simple one-to-one mapping
                resultValue = wsSource.Cells(rowNum, mappings(j).sourceCol).Value
            
            Case "STATIC"
                ' Static hardcoded value
                resultValue = mappings(j).staticValue
            
            Case "CONCAT"
                ' Concatenate multiple source columns
                cols = mappings(j).sourceCol
                resultValue = ""
                For k = LBound(cols) To UBound(cols)
                    If resultValue <> "" Then resultValue = resultValue & " "
                    resultValue = resultValue & wsSource.Cells(rowNum, cols(k)).Value
                Next k
            
            Case "FORMULA"
                ' Apply formula transformation
                sourceValue = wsSource.Cells(rowNum, mappings(j).sourceCol).Value
                resultValue = Evaluate(Replace(mappings(j).staticValue, "{SRC}", sourceValue))
            
            Case "CUSTOM"
                ' Call custom function
                resultValue = Application.Run(mappings(j).CustomFunction, wsSource, rowNum)
            
        End Select
        
        wsDestination.Cells(rowNum, mappings(j).destCol).Value = resultValue
    Next j
End Sub

Function CustomTransform_D3(wsSource As Worksheet, rowNum As Long) As Variant
    ' Example: Combine logic from multiple columns
    Dim val1 As Variant, val2 As Variant
    
    val1 = wsSource.Cells(rowNum, 6).Value  ' Example: read from column 6
    val2 = wsSource.Cells(rowNum, 7).Value  ' Example: read from column 7
    
    ' Apply your custom logic
    If IsNumeric(val1) And IsNumeric(val2) Then
        CustomTransform_D3 = CDbl(val1) + CDbl(val2)
    Else
        CustomTransform_D3 = "N/A"
    End If
End Function

Function ConditionalMapping_D9(wsSource As Worksheet, rowNum As Long) As Variant
    ' Example: Conditional logic based on source data
    Dim amount As Variant
    
    amount = wsSource.Cells(rowNum, 10).Value  ' Example: read amount from column 10
    
    ' Apply conditional logic
    If IsNumeric(amount) Then
        If CDbl(amount) > 10000 Then
            ConditionalMapping_D9 = "HIGH"
        ElseIf CDbl(amount) > 5000 Then
            ConditionalMapping_D9 = "MEDIUM"
        Else
            ConditionalMapping_D9 = "LOW"
        End If
    Else
        ConditionalMapping_D9 = "UNKNOWN"
    End If
End Function

Sub CreateTransformedSheet()
    Dim wsSource As Worksheet
    Dim wbNew As Workbook
    Dim wsDestination As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim savePath As String
    Dim fileName As String
    
    ' Set source worksheet (assumes the macro runs from the active sheet)
    Set wsSource = ActiveSheet
    
    ' Find last row with data in source sheet
    lastRow = wsSource.Cells(wsSource.Rows.Count, "A").End(xlUp).Row
    
    ' Create new workbook
    Set wbNew = Workbooks.Add
    Set wsDestination = wbNew.Worksheets(1)
    wsDestination.Name = "Transformed Data"
    
    ' Add headers to destination sheet
    wsDestination.Range("A1").Value = "D1"
    wsDestination.Range("B1").Value = "D2"
    wsDestination.Range("C1").Value = "D3"
    wsDestination.Range("D1").Value = "D4"
    wsDestination.Range("E1").Value = "D5"
    wsDestination.Range("F1").Value = "D6"
    wsDestination.Range("G1").Value = "D7"
    
    ' Copy and transform data (starting from row 2 to skip headers)
    For i = 2 To lastRow
        ' S5 ? D1 (Column A)
        wsDestination.Cells(i, 1).Value = wsSource.Cells(i, 5).Value
        
        ' S3 and S4 ? D2 (Column B) - Concatenated with space
        wsDestination.Cells(i, 2).Value = wsSource.Cells(i, 3).Value & " " & wsSource.Cells(i, 4).Value
        
        ' D3 (Column C) - Empty/Not mapped
        wsDestination.Cells(i, 3).Value = ""
        
        ' S2 ? D4 (Column D)
        wsDestination.Cells(i, 4).Value = wsSource.Cells(i, 2).Value
        
        ' S1 ? D5 (Column E)
        wsDestination.Cells(i, 5).Value = wsSource.Cells(i, 1).Value
        
        ' D6 (Column F) - Always "NEW"
        wsDestination.Cells(i, 6).Value = "NEW"
        
        ' D7 (Column G) - Always "CRED"
        wsDestination.Cells(i, 7).Value = "CRED"
    Next i
    
    ' Format headers
    With wsDestination.Range("A1:G1")
        .Font.Bold = True
        .Interior.Color = RGB(200, 200, 200)
    End With
    
    ' Auto-fit columns
    wsDestination.Columns("A:G").AutoFit
    
    ' Prompt user to save the new file
    fileName = "Transformed_" & Format(Now, "yyyymmdd_hhmmss") & ".xlsx"
    savePath = Application.GetSaveAsFilename( _
        InitialFileName:=fileName, _
        FileFilter:="Excel Files (*.xlsx), *.xlsx", _
        Title:="Save Transformed Data")
    
    ' Check if user cancelled the save dialog
    If savePath <> "False" Then
        wbNew.SaveAs fileName:=savePath, FileFormat:=xlOpenXMLWorkbook
        MsgBox "Transformation complete! New file saved as: " & vbCrLf & savePath, vbInformation
    Else
        MsgBox "File not saved. The new workbook is still open but unsaved.", vbExclamation
    End If
    
End Sub
